\section{LoadData}

Load Data Instruction

\subsection{Options}
\begin{description}
\item[file] = " "
\item[name] = " "
\item[makeCurrent] = yes
\end{description}

\subsection{Option Descriptions}
\begin{description}
\item[file] = " "
	\begin{description}
	\item[" "] = The file from which the data should be read
	\end{description}
\item[name] = " "
	\begin{description}
	\item[" "] = The dataset will be stored in the interpreter\'s map with the given name if name is not empty
\item[makeCurrent] = yes
\end{description}
