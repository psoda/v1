\section{Use Tree}

Use Tree Instruction

\subsection{Options}
\begin{description}
\item[name] = *default
\item[clearrep] = yes
\end{description}

\subsection{Option Descriptions}
\begin{description}
\item[name] = *default
	\begin{description}
	\item[*default] = The name of the tree to use
	\end{description}
\item[clearrep] = yes
	\begin{description}
	\item[yes] = If this parameter is 'yes', clears the contents of the tree repository before adding the tree
	\end{description}
\end{description}
